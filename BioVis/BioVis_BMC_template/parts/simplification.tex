\begin{algorithm}
  \begin{algorithmic}[1]
	  \Require $T$ --- trajectory, $S$ --- simplification
	  \Ensure $T$ --- simplified trajectory
		\Procedure{Simplify}{$T, S$}
			\State $S' \gets$ \Call{CacheLoad}{$ $} \Comment{$S'$ --- previous simplification}
			\State
			\If{\Call{IsDiffIncremental}{$S, S'$}}
			  \State $\Delta I \gets$ \Call{Diff}{$S, S'$}
				\State $T \gets$ \Call{SavitzkyGolay}{$T, \Delta I$}
			\Else %\Comment{compute from scratch}
			  \State $\mathcal{L} \gets$ \Call{ByLevels}{$S$} \Comment{$\mathcal{L}$ --- sets of complex intervals}
				\ForAll {$L \in \mathcal{L}$} \Comment{in asc. order by $level(L)$}
				  \ForAll {$I \in L$}
					  \State $T \gets$ \Call{SavitzkyGolay}{$T, I$}
					\EndFor
				\EndFor
			\EndIf
			\State
			\State \Call{CacheSave}{$S$}
			\State \Return $T$
		\EndProcedure
  \end{algorithmic}
	\caption{Trajectory simplification}
  \label{alg:simplify}
\end{algorithm}

\begin{algorithm}
  \begin{algorithmic}[1]
	  \Require $T$ --- trajectory, $\nu$ --- complexity neighborhood, $\tau$ --- complexity threshold, $\epsilon$ --- improvement threshold
	  \Ensure $T$ --- simplified trajectory
		\Procedure{AutoSimplify}{$T, \nu, \tau, \epsilon$}
			\State $C \gets$ \Call{Interval}{$T$} \Comment{$C$ --- complex intervals}
			\State $c(x) \gets$ \Call{Complexity}{$T, \nu$}
			\State
			\Repeat
			  \State $S \gets$ \Call{FindSimpleIntervals}{$c, \tau$}
				\State $C \gets$ \Call{RemoveSimpleIntervals}{$C, S$}
			  \State
			  \ForAll{$I \in C$} %\Comment{$I$ --- complex interval}
					\State $T \gets$ \Call{Simplify}{$T, I$}
			  \EndFor
				\State
				\State $c'(x) \gets c(x)$ %\Comment{$c'(x)$ --- previous complexity}
				\State $c(x) \gets$ \Call{Complexity}{$T, \nu$}
				\State
				\State $\Delta c \gets \sum_{x \in T}{max(c(x) - c'(x), 0)}$
				\State \Comment{$\Delta c$ --- improvement}
			\Until{$\Delta c < \epsilon$}
			\State
			\State \Return $T$
		\EndProcedure
  \end{algorithmic}
	\caption{Automatic trajectory simplification}
  \label{alg:auto-simplify}
\end{algorithm}
